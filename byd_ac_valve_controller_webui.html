// ESP32-S3 Arduino — WiFi Direct + Smart Flow Sequence
#include <WiFi.h>
#include <WebServer.h>

WebServer server(80);
const char* ssid     = "BYD_AC_CONTROLLER";
const char* password = "12345678";
const int   VALVE_PINS[6] = {12, 13, 14, 25, 26, 27};

void setup() {
  for (int i = 0; i < 6; i++) { pinMode(VALVE_PINS[i], OUTPUT); digitalWrite(VALVE_PINS[i], LOW); }
  WiFi.softAP(ssid, password);

  server.on("/start",     []{ server.send(200, "text/plain", "SYSTEM START"); });
  server.on("/stop",      []{ allOff(); server.send(200, "text/plain", "SYSTEM STOP"); });
  server.on("/emergency", []{ allOff(); server.send(200, "text/plain", "EMERGENCY STOP"); });

  // Individual valve: /valve?id=0&state=1
  server.on("/valve", []() {
    int id = server.arg("id").toInt(), state = server.arg("state").toInt();
    if (id >= 0 && id < 6) {
      digitalWrite(VALVE_PINS[id], state ? HIGH : LOW);
      server.send(200, "text/plain", "V" + String(id+1) + (state ? " ON" : " OFF"));
    }
  });

  // Multi-valve: /valves?mask=0b101010  (bit 0 = V1 ... bit 5 = V6)
  server.on("/valves", []() {
    int mask = server.arg("mask").toInt();
    for (int i = 0; i < 6; i++) digitalWrite(VALVE_PINS[i], (mask >> i) & 1 ? HIGH : LOW);
    server.send(200, "text/plain", "MASK:" + String(mask, BIN));
  });

  server.begin();
}

void loop()  { server.handleClient(); }
void allOff(){ for (int i = 0; i < 6; i++) digitalWrite(VALVE_PINS[i], LOW); }
